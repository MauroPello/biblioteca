SELECT libri.*, prestiti.ID_tessera_utenti, prestiti.data_inizio FROM libri INNER JOIN prestiti ON libri.ID_codiceISBN = prestiti.ID_codiceISBN_libri WHERE (prestiti.data_fine IS NULL AND prestiti.data_inizio <= '*Data Attuale Meno Due Mesi*');
SELECT tabellaconteggi.ID_tessera_utenti, tabellaconteggi.ID_codiceISBN_libri, tabellaconteggi.conteggiolibrinoleggiati FROM (SELECT ID_tessera_utenti, MAX(conteggiolibrinoleggiati) AS massimo FROM (SELECT prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri, COUNT(prestiti.ID_codiceISBN_libri) AS conteggiolibrinoleggiati FROM prestiti GROUP BY prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri)  AS [%$##@_Alias] GROUP BY ID_tessera_utenti)  AS tabellamassimi INNER JOIN (SELECT prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri, COUNT(prestiti.ID_codiceISBN_libri) AS conteggiolibrinoleggiati FROM prestiti GROUP BY prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri)  AS tabellaconteggi ON (tabellaconteggi.conteggiolibrinoleggiati = tabellamassimi.massimo) AND (tabellaconteggi.ID_tessera_utenti = tabellamassimi.ID_tessera_utenti);
SELECT tabellaconteggi.nome, tabellaconteggi.ID_codiceISBN_libri, tabellaconteggi.conteggio FROM (SELECT nome, MAX(conteggio) AS massimo FROM (SELECT prestiti.ID_codiceISBN_libri, categorie.nome, count(prestiti.ID_codiceISBN_libri) AS conteggio FROM (libri INNER JOIN (categorie INNER JOIN libricategorie ON categorie.ID = libricategorie.ID_categorie) ON libri.ID_codiceISBN = libricategorie.ID_codiceISBN_libri) INNER JOIN prestiti ON libri.ID_codiceISBN = prestiti.ID_codiceISBN_libri GROUP BY prestiti.ID_codiceISBN_libri, categorie.nome)  AS [%$##@_Alias] GROUP BY nome)  AS tabellamassimi INNER JOIN (SELECT prestiti.ID_codiceISBN_libri, categorie.nome, count(prestiti.ID_codiceISBN_libri) AS conteggio FROM (libri INNER JOIN (categorie INNER JOIN libricategorie ON categorie.ID = libricategorie.ID_categorie) ON libri.ID_codiceISBN = libricategorie.ID_codiceISBN_libri) INNER JOIN prestiti ON libri.ID_codiceISBN = prestiti.ID_codiceISBN_libri GROUP BY prestiti.ID_codiceISBN_libri, categorie.nome)  AS tabellaconteggi ON (tabellaconteggi.nome = tabellamassimi.nome) AND (tabellaconteggi.conteggio = tabellamassimi.massimo);
SELECT categorie.nome, autori.nome, autori.cognome FROM (libri INNER JOIN (autori INNER JOIN libriautori ON autori.ID = libriautori.ID_autori) ON libri.ID_codiceISBN = libriautori.ID_codiceISBN_libri) INNER JOIN (categorie INNER JOIN libricategorie ON categorie.ID = libricategorie.ID_categorie) ON libri.ID_codiceISBN = libricategorie.ID_codiceISBN_libri WHERE (((categorie.nome)='*Nome*'));
SELECT tabellaconteggi.ID_tessera_utenti, autori.nome, autori.cognome, tabellaconteggi.conteggio FROM autori INNER JOIN ((SELECT ID_tessera_utenti, MAX(conteggio) AS massimo FROM (SELECT ID_tessera_utenti, libriautori.ID_autori, SUM(tabellaconteggiprestiti.conteggioprestiti) AS conteggio FROM (libri INNER JOIN libriautori ON libri.ID_codiceISBN = libriautori.ID_codiceISBN_libri) INNER JOIN (SELECT prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri, COUNT(prestiti.ID_codiceISBN_libri) AS conteggioprestiti FROM prestiti GROUP BY prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri)  AS tabellaconteggiprestiti ON libri.ID_codiceISBN = tabellaconteggiprestiti.ID_codiceISBN_libri GROUP BY ID_tessera_utenti, libriautori.ID_autori)  AS [%$##@_Alias] GROUP BY ID_tessera_utenti)  AS tabellamassimi INNER JOIN (SELECT ID_tessera_utenti, libriautori.ID_autori, SUM(tabellaconteggiprestiti.conteggioprestiti) AS conteggio FROM (libri INNER JOIN libriautori ON libri.ID_codiceISBN = libriautori.ID_codiceISBN_libri) INNER JOIN (SELECT prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri, COUNT(prestiti.ID_codiceISBN_libri) AS conteggioprestiti FROM prestiti GROUP BY prestiti.ID_tessera_utenti, prestiti.ID_codiceISBN_libri)  AS tabellaconteggiprestiti ON libri.ID_codiceISBN = tabellaconteggiprestiti.ID_codiceISBN_libri GROUP BY ID_tessera_utenti, libriautori.ID_autori)  AS tabellaconteggi ON (tabellamassimi.massimo = tabellaconteggi.conteggio) AND (tabellamassimi.ID_tessera_utenti = tabellaconteggi.ID_tessera_utenti)) ON autori.ID = tabellaconteggi.ID_autori;
SELECT libri.*, categorie.nome, prestiti.data_fine, utenti.nome, utenti.cognome FROM utenti INNER JOIN ((libri INNER JOIN (categorie INNER JOIN libricategorie ON categorie.ID = libricategorie.ID_categorie) ON libri.ID_codiceISBN=libricategorie.[ID_codiceISBN_libri]) INNER JOIN prestiti ON libri.ID_codiceISBN=prestiti.[ID_codiceISBN_libri]) ON utenti.[ID_tessera]=prestiti.[ID_tessera_utenti] WHERE (((categorie.nome)='*Nome*') AND ((prestiti.data_fine) Is Not Null) AND ((utenti.ID_tessera)='*ID_Tessera*'));
SELECT libri.*, libriautori.*, autori.nome, autori.cognome FROM libri INNER JOIN (autori INNER JOIN libriautori ON autori.ID = libriautori.ID_autori) ON libri.ID_codiceISBN=libriautori.[ID_codiceISBN_libri] WHERE (((autori.nome)='*Nome*') AND ((autori.cognome)='*Cognome*'));
